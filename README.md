# Clone this repository :
- git clone https://github.com/imtiaj-007/SpeechToText.git
- git clone git@github.com:imtiaj-007/SpeechToText.git

# Install Dependencies:
    npm install

# Run Project:
    npm run dev

# Build Project:
    vite build
